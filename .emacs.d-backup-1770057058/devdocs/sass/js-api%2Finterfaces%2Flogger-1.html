Logger

An object that can be passed to logger to control how Sass emits warnings and debug messages.

Example

constfs = require('fs');
constsass = require('sass');

letlog = "";
sass.renderSync({
file:'input.scss',
logger: {
warn(message, options) {
if (options.span) {
log += `${span.url}:${span.start.line}:${span.start.column}: ` +
`${message}\n`;
      } else {
log += `::: ${message}\n`;
      }
    }
  }
});

fs.writeFileSync('log.txt', log);
