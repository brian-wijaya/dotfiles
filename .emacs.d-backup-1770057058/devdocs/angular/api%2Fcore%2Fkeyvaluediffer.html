KeyValueDiffer

A differ that tracks changes made to an object over time.

API

interface KeyValueDiffer<K, V> {
  diff(object: Map<K, V>): KeyValueChanges<K, V> | null;
  diff(object: { [key: string]: V; }): KeyValueChanges<string, V> | null;
}
